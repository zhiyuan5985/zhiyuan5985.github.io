1. 实现(5).add(3).minus(2)

比较好的写法：

function check(n) {
    // 检验是否有效数字,参数的容错性
    n = Number(n);                
    return isNaN(n) ? 0 : n;
}
~function () {
    function add (n) {
        n = check(n) 
        return this + n;
    }
       function minus (n) {
        n = check(n)
        return this - n;
    }
    Number.prototype.add = add;
    Number.prototype.minus = minus;
}();
console.log((5).add(3).minus(2));  // 返回6

2. 箭头函数和普通函数function的区别是什么？构造函数可以用new生成实例，箭头函数可以吗？

（1）箭头函数在语法上更加简洁

（2）箭头函数当中没有自己的this，它里面出现的this继承所处上下文的this，使用call/apply等任何方式都不能改变this指向

document.body.onclick = () => {
    // THIS指向Window，而不是当前操作的BODY
}
document.body.onclick = function () {
    // THIS指向BODY
    arr.sort(function(a, b) {
        // 这里是回调函数，一般指向Window
        // 回调函数：把一个函数B作为实参传给另外一个函数A
        return a - b; 
    })
    arr.sort((a, b) => {
        // 箭头函数THIS指向上下文，此处是BODY
        return a - b; 
    })
}

（3）箭头函数当中没有arguments，但是可以用…arg

（4）箭头函数不能被new执行，因为箭头函数没有prototype

let arr = [10, 20, 30, 'A', 40],
    obj = {}
Array.prototype.each = function(callback, obj) {
    if (obj !== null) {
        callback.call(obj);
    }
    for (let [index, item] of this.entries()) {
        flag = callback(item, index);
        if (!flag) {
            break;
        } else {
            this[index] = flag;
        }
    }
}

arr.each(function(item, index) {
    // this指向obj，若没有obj指向window
    if (isNaN(item)) {
        return false;
    }
    return item * 10;
}, obj)

3. 如何把字符串的大小写取反，例如’Abc’变成’aBc’

用正则匹配的办法比较好：

let str = "Zhiyuan is very Handsome,HeyHey ^_^"
str = str.replace(/[a-zA-Z]/g, content => {
    // content是每一次正则匹配的结果
    // 验证是否是大写字母：把字母转换成大写后看和之前是否一样，如果一样，之前也是大写的;
    // 或者使用ASCII码范围判断
    return content === content.toUpperCase() ? content.toLowerCase() : content.toUpperCase();
});
console.log(str);

4. 关于对象中的键

example 1：

var a = {}, b = '123', c = 123;
a[b] = 'b';
a[c] = 'c'
console.log(a[b]); // => c
// 因为对象key中的数字123和字符串123没有区别，所以后面把前面顶掉了。

example 2：

var a = {}, b = Symbol('123'), c = Symbol('123');
a[b] = 'b';
a[c] = 'c'
console.log(a[b]);  // => b
// Symbol是ES6中新增的数据类型，创建的是唯一值。

example 3：

var a = {}, b = {key: '123'}, c = {key: '456'};
a[b] = 'b';
a[c] = 'c';
console.log(a[b]);    // => c
// 因为对象作为键值最后都会转换为“[object Object]”

5. 验证用户输入是否符合URL网址的规则

let str = "http://www.zhufengpeixun.cn/index.html?lx=asaasascd";
// URL格式：
// 1.协议    2.域名部分（www.xxx.xx 或者 xxx.xx 或者 kbs.sports.qq.com.cn类似的）不能省
// 3.请求路径 4.问号传参 5.哈希值
// i表示忽略大小写匹配,域名：数字字母下划线加-
let reg = /^((http|https|ftp):\/\/)?(([\w-]+\.)+[a-Z0-9]+)((\/[^/]*)+)?(\?[^#]+)?(#.+)$/i;  
console.log(reg.exec(str))

6. 关于构造函数

function Foo() {
     Foo.a = function () {
        console.log(1);
    }
    this.a = function () {
        console.log(2);
    }
}
// 把Foo当作类，在原型上设置共有的属性方法
Foo.prototype.a = function () {
    console.log(3);
}
// 把Foo当作普通对象设置私有属性方法
Foo.a = function () {
    console.log(4);
}
Foo.a();    // =>4 把Foo当作普通对象
let obj = new Foo();  // obj可以调取原型上的方法，此时Foo()会执行
obj.a();    // =>2 obj对象调取原型上方法
Foo.a();    // =>1
